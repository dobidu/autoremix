  <files>plugin/include/SidecarHealthDot.h</files>
  <action>
    Create a custom component that polls a provided function and paints a dot:

    ```cpp
    #pragma once
    #include <juce_gui_basics/juce_gui_basics.h>
    #include <juce_gui_extra/juce_gui_extra.h>
    #include "AutoRemixLookAndFeel.h"
    #include <functional>

    class SidecarHealthDot : public juce::Component, private juce::Timer {
    public:
        explicit SidecarHealthDot(std::function<bool()> pollFn)
            : poll_fn_(std::move(pollFn))
        {
            setTooltip("Sidecar: checking...");
            startTimer(2000);
            timerCallback();  // immediate first poll
        }

        ~SidecarHealthDot() override { stopTimer(); }

        void paint(juce::Graphics& g) override {
            auto bounds = getLocalBounds().toFloat().reduced(1.0f);
            switch (state_) {
                case State::connected:    g.setColour(juce::Colour(AR::GREEN));  break;
                case State::disconnected: g.setColour(juce::Colour(AR::RED));    break;
                case State::checking:     g.setColour(juce::Colour(AR::ORANGE)); break;
            }
            g.fillEllipse(bounds);
        }

    private:
        enum class State { checking, connected, disconnected };

        std::function<bool()> poll_fn_;
        State state_ = State::checking;

        void timerCallback() override {
            bool alive = poll_fn_();
            auto newState = alive ? State::connected : State::disconnected;
            if (newState != state_) {
                state_ = newState;
                setTooltip(alive ? "Sidecar: connected" : "Sidecar: disconnected");
                repaint();
            }
        }

        JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(SidecarHealthDot)
    };
    ```

    Note: `juce_gui_extra` is required for `juce::TooltipWindow`; if not in module list, use `setHelpText` or omit tooltip (just remove the setTooltip calls). Check CMakeLists.txt first.

<task type="auto">
  <name>Task 2: Create WaveformDisplay.h</name>
  <files>plugin/include/WaveformDisplay.h</files>
  <action>
    Create a dedicated component for the waveform zone. Area fill is achieved by calling
    drawChannels twice: once at 15% alpha (area fill illusion) then again at 100% (crisp waveform).

    ```cpp
    #pragma once
    #include <juce_audio_utils/juce_audio_utils.h>
    #include <juce_gui_basics/juce_gui_basics.h>
    #include "AutoRemixLookAndFeel.h"

    class WaveformDisplay : public juce::Component {
    public:
        explicit WaveformDisplay(juce::AudioThumbnail& thumb) : thumb_(thumb) {
            setBufferedToImage(true);
        }

        void paint(juce::Graphics& g) override {
            auto area = getLocalBounds();

            g.setColour(juce::Colour(AR::BG_DEEP));
            g.fillRoundedRectangle(area.toFloat(), 4.0f);

            if (thumb_.getTotalLength() > 0.0) {
                double len = thumb_.getTotalLength();

                // area fill: 15% alpha under the waveform shape
                g.setColour(juce::Colour(AR::PURPLE).withAlpha(0.15f));
                thumb_.drawChannels(g, area, 0.0, len, 1.0f);

                // crisp waveform on top
                g.setColour(juce::Colour(AR::PURPLE));
                thumb_.drawChannels(g, area, 0.0, len, 1.0f);
            } else {
                g.setFont(AR::font(AR::FontRole::secondary));
                g.setColour(juce::Colour(AR::COMMENT));
                g.drawFittedText("No file loaded", area, juce::Justification::centred, 1);
            }
        }

        // Call when thumbnail source changes so buffered image refreshes
        void sourceChanged() {
            setBufferedToImage(false);
            setBufferedToImage(true);
            repaint();
        }

    private:
        juce::AudioThumbnail& thumb_;

        JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(WaveformDisplay)
    };
    ```
