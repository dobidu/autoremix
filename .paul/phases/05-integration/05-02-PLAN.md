    Replace `bool AudioBridge::startSidecar(const std::filesystem::path&) { return false; }`:
    ```cpp
    bool AudioBridge::startSidecar(const std::filesystem::path& server_script_path) {
        if (sidecar_pid_ > 0) return true;   // already running

        pid_t pid = fork();
        if (pid < 0) return false;

        if (pid == 0) {
            // child: exec uvicorn or python server/main.py
            const char* python = "python3";
            std::string script = server_script_path.string();
            execl("/usr/bin/python3", python, script.c_str(), nullptr);
            // fallback: search PATH
            execlp("python3", python, script.c_str(), nullptr);
            _exit(127);
        }

        sidecar_pid_ = pid;

        // poll isServerAlive for up to 10s
        using clock = std::chrono::steady_clock;
        auto deadline = clock::now() + std::chrono::seconds(10);
        while (clock::now() < deadline) {
            if (isServerAlive()) return true;
            std::this_thread::sleep_for(std::chrono::milliseconds(250));
        }
        // timed out — kill child and reset
        kill(sidecar_pid_, SIGKILL);
        waitpid(sidecar_pid_, nullptr, 0);
        sidecar_pid_ = -1;
        return false;
    }
    ```

    Replace `void AudioBridge::stopSidecar() {}`:
    ```cpp
    void AudioBridge::stopSidecar() {
        if (sidecar_pid_ <= 0) return;
        kill(sidecar_pid_, SIGTERM);
        // give it 2s to exit gracefully, then SIGKILL
        for (int i = 0; i < 8; ++i) {
            std::this_thread::sleep_for(std::chrono::milliseconds(250));
            if (waitpid(sidecar_pid_, nullptr, WNOHANG) == sidecar_pid_) {
                sidecar_pid_ = -1;
                return;
            }
        }
        kill(sidecar_pid_, SIGKILL);
        waitpid(sidecar_pid_, nullptr, 0);
        sidecar_pid_ = -1;
    }
    ```

    Note: `execl` path `/usr/bin/python3` covers Ubuntu CI; `execlp` fallback covers other paths. On macOS the path differs — `execlp` fallback handles that.
    Avoid: using `system()` (shell injection risk); avoid `popen` (no PID access for clean kill).
  </action>
