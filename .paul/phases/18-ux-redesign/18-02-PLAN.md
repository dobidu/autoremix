```cpp
#pragma once
#include <juce_gui_basics/juce_gui_basics.h>
#include "ScreenContext.h"

// ─────────────────────────────────────────────────────────────────────────────
// ScreenBase — abstract base for all AutoRemix screens.
//
// Each screen is a juce::Component that fills the main content area (960×520 px,
// below the 48px header and above the 32px footer).
//
// Lifecycle:
//   1. navigateTo(id) constructs the new ScreenBase-derived object.
//   2. Editor calls addAndMakeVisible(*screen_) and sets its bounds.
//   3. Editor calls screen_->onEnter() — load data, start timers, etc.
//   4. When navigating away: screen_->onExit() → fade out → destroy.
//
// Convention: screens must NOT start background threads in their constructor;
// use onEnter() instead so the component is visible before work begins.
// ─────────────────────────────────────────────────────────────────────────────
class ScreenBase : public juce::Component {
public:
    explicit ScreenBase(ScreenContext& ctx) : ctx_(ctx) {}
    ~ScreenBase() override = default;

    // Called immediately after the screen is added and its bounds are set.
    // Override to initialize state, load thumbnails, start timers, etc.
    virtual void onEnter() {}

    // Called just before the screen is removed and destroyed.
    // Override to stop timers, cancel background work cleanly.
    virtual void onExit() {}

protected:
    ScreenContext& ctx_;

private:
    JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(ScreenBase)
};
```

```cpp
void AutoRemixAudioProcessorEditor::navigateTo(ScreenId id, bool animate)
{
    // Must be called on the message thread
    jassert(juce::MessageManager::getInstance()->isThisTheMessageThread());

    // ── Tear down old screen ──────────────────────────────────────────────────
    if (screen_) {
        screen_->onExit();
        if (animate) {
            // Fade old screen out, then swap
            auto* oldScreen = screen_.release();
            auto& animator  = juce::Desktop::getInstance().getAnimator();
            animator.animateComponent(oldScreen, oldScreen->getBounds(), 0.0f, 150,
                                      false, 1.0, 1.0);
            // Use a timer to destroy after fade and install new screen
            juce::Timer::callAfterDelay(160, [this, oldScreen, id]() mutable {
                removeChildComponent(oldScreen);
                delete oldScreen;
                installScreen(id, /*animate=*/true);
            });
            return;
        } else {
            removeChildComponent(screen_.get());
            screen_.reset();
        }
    }

    installScreen(id, animate);
}
```

```cpp
void AutoRemixAudioProcessorEditor::installScreen(ScreenId id, bool animate)
{
    // ── Construct new screen ──────────────────────────────────────────────────
    std::unique_ptr<ScreenBase> newScreen;
    switch (id) {
        case ScreenId::Empty:      newScreen = std::make_unique<ScreenEmptyStub>(ctx_); break;
        case ScreenId::Separating: newScreen = std::make_unique<ScreenEmptyStub>(ctx_); break;
        case ScreenId::StemsReady: newScreen = std::make_unique<ScreenEmptyStub>(ctx_); break;
        case ScreenId::ModeParams: newScreen = std::make_unique<ScreenEmptyStub>(ctx_); break;
        case ScreenId::Render:     newScreen = std::make_unique<ScreenEmptyStub>(ctx_); break;
        default:                   newScreen = std::make_unique<ScreenEmptyStub>(ctx_); break;
    }

    // ── Layout: main area is below 48px header, above 32px footer ────────────
    // Total height 600: header=48, footer=32, content=520
    auto contentBounds = juce::Rectangle<int>(0, 48, 960, 520);
    newScreen->setBounds(contentBounds);

    screen_ = std::move(newScreen);
    addAndMakeVisible(*screen_);

    if (animate) {
        // Start fully transparent and animate to opaque
        screen_->setAlpha(0.0f);
        auto& animator = juce::Desktop::getInstance().getAnimator();
        animator.animateComponent(screen_.get(), contentBounds, 1.0f, 150, false, 1.0, 1.0);
    } else {
        screen_->setAlpha(1.0f);
    }

    screen_->onEnter();
}
```
