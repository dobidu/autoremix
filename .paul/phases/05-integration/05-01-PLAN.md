    from pathlib import Path

    async def test_separate_algorithmic(client, test_wav, tmp_path):

    **python/tests/test_remix.py:**
    ```python
    import pytest
    from pathlib import Path

    # Shared stems fixture — separate once, remix 3 times
    @pytest.fixture(scope="module")
    async def stems(client, test_wav, tmp_path_factory):
        out = tmp_path_factory.mktemp("stems")
        r = await client.post("/api/v1/separate", json={
            "input_path": str(test_wav),
            "output_dir": str(out),
            "separator_id": "algorithmic",
        })
        assert r.json()["success"] is True
        return r.json()["stems"]

    @pytest.mark.parametrize("engine_id", [
        "chopped_screwed", "slowed_reverb", "drum_and_bass"
    ])
    async def test_remix_engine(client, stems, tmp_path_factory, engine_id):
        out = tmp_path_factory.mktemp("remix") / f"{engine_id}_out.wav"
        r = await client.post("/api/v1/remix", json={
            "vocals_path": stems["vocals"],
            "drums_path":  stems["drums"],
            "bass_path":   stems["bass"],
            "other_path":  stems["other"],
            "output_path": str(out),
            "engine_id":   engine_id,
        })
        assert r.status_code == 200
        data = r.json()
        assert data["success"] is True, f"engine {engine_id} failed: {data.get('error')}"
        assert Path(data["output_path"]).exists()
        assert Path(data["output_path"]).stat().st_size > 0

    async def test_remix_unknown_engine(client, stems, tmp_path):
        r = await client.post("/api/v1/remix", json={
            "vocals_path": stems["vocals"],
            "drums_path":  stems["drums"],
            "bass_path":   stems["bass"],
            "other_path":  stems["other"],
            "output_path": str(tmp_path / "out.wav"),
            "engine_id":   "does_not_exist",
        })
        assert r.status_code == 400
    ```

    Note on `stems` fixture: `scope="module"` + async — pytest-asyncio handles this
    when `asyncio_mode = auto`. The `client` fixture must also be module-scoped for
    this to work. Change `client` fixture in conftest to `scope="module"` if needed,
    or use a separate `module_client` fixture. Simplest: keep both at function scope
    and accept that separation runs once per remix test (3s synthetic WAV is fast <1s).
    Remove `scope="module"` from stems fixture if it causes scope mismatch.

    Current issues:
    - Unexpected exceptions in separate/remix return 500 (unhandled Exception → FastAPI
      default). Already caught in the except block but need to verify no edge cases slip.
    - KeyError for unknown separator/engine already returns 400 via HTTPException — good.
    - Missing input_path: already returns 400 via HTTPException — good.
    - Edge case: `separator.separate()` raises something other than Exception child
      that's not caught → 500. Unlikely but harden.

    Verify the current except blocks catch broadly enough:
    ```python
    except KeyError as e:
        raise HTTPException(status_code=400, detail=f"Unknown id: {e}")
    except Exception as e:
        logger.exception("...")
        return SeparateResponse(success=False, error=str(e))
    ```

    If already correct, no changes needed — mark task done.
    If not, tighten so:
    - `KeyError` → 400 HTTPException (already done)
    - `HTTPException` re-raised (not swallowed by broad `except Exception`)
    - All other exceptions → `success=False, error=str(e)` in response body

    Critical: `except Exception` must NOT catch `HTTPException` (which is a Starlette
    exception, subclass of Exception). Add explicit re-raise:
    ```python
    except HTTPException:
        raise
    except KeyError as e:
        raise HTTPException(status_code=400, detail=f"Unknown id: {e}")
    except Exception as e:
        logger.exception("...")
        return SeparateResponse(success=False, error=str(e))
    ```
    Apply same pattern to both `/separate` and `/remix` handlers.
