Create `plugin/include/ScreenStemsReady.h` with the following full implementation:

```cpp
#pragma once
#include <juce_audio_utils/juce_audio_utils.h>
#include <juce_gui_basics/juce_gui_basics.h>
#include <array>
#include "ScreenBase.h"
#include "AutoRemixLookAndFeel.h"
#include "PluginTypes.h"

class ScreenStemsReady : public ScreenBase,
                         public juce::ChangeListener
{
public:
    explicit ScreenStemsReady(ScreenContext& ctx)
        : ScreenBase(ctx),
          thumbnail_cache_(8),
          vocals_thumb_(512, format_manager_, thumbnail_cache_),
          drums_thumb_ (512, format_manager_, thumbnail_cache_),
          bass_thumb_  (512, format_manager_, thumbnail_cache_),
          other_thumb_ (512, format_manager_, thumbnail_cache_)
    {
        format_manager_.registerBasicFormats();

        vocals_thumb_.addChangeListener(this);
        drums_thumb_ .addChangeListener(this);
        bass_thumb_  .addChangeListener(this);
        other_thumb_ .addChangeListener(this);

        // Initialise per-stem rows from spec
        // rows_[i] = { name, color_token, *ctx_gain, *ctx_muted, *ctx_solo, thumb_ref }
        initRows();

        // Action bar buttons
        addAndMakeVisible(back_btn_);
        back_btn_.setButtonText("\xe2\x86\x90 Back");  // ← UTF-8
        back_btn_.setComponentID("ghost");
        back_btn_.onClick = [this] { ctx_.navigate(ScreenId::Empty); };

        addAndMakeVisible(next_btn_);
        next_btn_.setButtonText("Choose Style \xe2\x86\x92");  // → UTF-8
        next_btn_.setComponentID("primary");
        next_btn_.onClick = [this] { ctx_.navigate(ScreenId::ModeParams); };
    }

    ~ScreenStemsReady() override
    {
        vocals_thumb_.removeChangeListener(this);
        drums_thumb_ .removeChangeListener(this);
        bass_thumb_  .removeChangeListener(this);
        other_thumb_ .removeChangeListener(this);
    }

    void onEnter() override
    {
        // Load stem thumbnails from ctx_ paths
        if (ctx_.stems.valid) {
            auto loadThumb = [](juce::AudioThumbnail& thumb, const std::filesystem::path& p) {
                juce::File f(juce::String(p.string()));
                if (f.existsAsFile())
                    thumb.setSource(new juce::FileInputSource(f));
            };
            loadThumb(vocals_thumb_, ctx_.stems.vocals);
            loadThumb(drums_thumb_,  ctx_.stems.drums);
            loadThumb(bass_thumb_,   ctx_.stems.bass);
            loadThumb(other_thumb_,  ctx_.stems.other);
        }
        resized();  // re-layout in case bounds changed
        repaint();
    }

    void onExit() override {}

    void changeListenerCallback(juce::ChangeBroadcaster*) override { repaint(); }

    void paint(juce::Graphics& g) override
    {
        g.fillAll(juce::Colour(AR::BG));

        // Section header
        auto b = getLocalBounds().withTrimmedBottom(72);
        auto headerRow = b.removeFromTop(40);
        g.setFont(AR::font(AR::FontRole::section_label));
        g.setColour(juce::Colour(AR::COMMENT));
        g.drawText("STEMS", headerRow.reduced(16, 0), juce::Justification::centredLeft);

        // Paint waveform for each row — buttons/sliders are child Components
        for (int i = 0; i < 4; ++i) {
            auto row = b.removeFromTop(ROW_H);
            paintRowBackground(g, row, i);
        }
    }

    void resized() override
    {
        auto b = getLocalBounds();
        auto actionBar = b.removeFromBottom(72);
        b.removeFromTop(40);  // section header

        for (int i = 0; i < 4; ++i) {
            auto row = b.removeFromTop(ROW_H);
            layoutRow(row, i);
        }

        // Action bar
        actionBar = actionBar.reduced(16, 16);
        back_btn_.setBounds(actionBar.removeFromLeft(120));
        next_btn_.setBounds(actionBar.removeFromRight(180));
    }

private:
    static constexpr int ROW_H        = 56;
    static constexpr int LEFT_COL_W   = 156;
    static constexpr int RIGHT_COL_W  = 280;
    static constexpr int TOGGLE_SIZE  = 24;
    static constexpr int TOGGLE_GAP   = 4;
    static constexpr int SLIDER_W     = 220;

    struct StemRowWidgets {
        juce::String        name;
        uint32_t            color;
        juce::AudioThumbnail* thumb = nullptr;
        juce::TextButton    mute_btn;
        juce::TextButton    solo_btn;
        juce::Slider        gain_slider;
        float*              ctx_gain   = nullptr;
        bool*               ctx_muted  = nullptr;
        bool*               ctx_solo   = nullptr;
    };

    std::array<StemRowWidgets, 4> rows_;
    bool rows_initialized_ = false;

    void initRows()
    {
        auto setup = [&](int i, const char* name, uint32_t color,
                         juce::AudioThumbnail& thumb,
                         float& gain, bool& muted, bool& solo) {
            auto& r = rows_[i];
            r.name        = name;
            r.color       = color;
            r.thumb       = &thumb;
            r.ctx_gain    = &gain;
            r.ctx_muted   = &muted;
            r.ctx_solo    = &solo;

            // Mute button
            addAndMakeVisible(r.mute_btn);
            r.mute_btn.setButtonText("M");
            r.mute_btn.setClickingTogglesState(true);
            r.mute_btn.setColour(juce::TextButton::buttonOnColourId, juce::Colour(AR::ERROR));
            r.mute_btn.onClick = [this, i] { handleMute(i); };

            // Solo button
            addAndMakeVisible(r.solo_btn);
            r.solo_btn.setButtonText("S");
            r.solo_btn.setClickingTogglesState(true);
            r.solo_btn.setColour(juce::TextButton::buttonOnColourId, juce::Colour(AR::WARNING));
            r.solo_btn.onClick = [this, i] { handleSolo(i); };

            // Gain slider
            addAndMakeVisible(r.gain_slider);
            r.gain_slider.setSliderStyle(juce::Slider::LinearHorizontal);
            r.gain_slider.setTextBoxStyle(juce::Slider::NoTextBox, true, 0, 0);
            r.gain_slider.setRange(0.0, 2.0, 0.01);
            r.gain_slider.setValue(gain, juce::dontSendNotification);
            r.gain_slider.onValueChange = [this, i] {
                *rows_[i].ctx_gain = (float)rows_[i].gain_slider.getValue();
            };
        };

        setup(0, "Vocals", AR::STEM_VOCALS, vocals_thumb_, ctx_.vocals_gain, ctx_.vocals_muted, ctx_.vocals_solo);
        setup(1, "Drums",  AR::STEM_DRUMS,  drums_thumb_,  ctx_.drums_gain,  ctx_.drums_muted,  ctx_.drums_solo);
        setup(2, "Bass",   AR::STEM_BASS,   bass_thumb_,   ctx_.bass_gain,   ctx_.bass_muted,   ctx_.bass_solo);
        setup(3, "Other",  AR::STEM_OTHER,  other_thumb_,  ctx_.other_gain,  ctx_.other_muted,  ctx_.other_solo);
        rows_initialized_ = true;
    }

    void paintRowBackground(juce::Graphics& g, juce::Rectangle<int> row, int i)
    {
        if (!rows_initialized_) return;
        auto& r = rows_[i];

        // Subtle alternating row bg
        if (i % 2 == 1) {
            g.setColour(juce::Colour(AR::ELEVATED).withAlpha(0.4f));
            g.fillRect(row);
        }

        // Divider at bottom
        g.setColour(juce::Colour(AR::SURFACE));
        g.fillRect(row.removeFromBottom(1));

        // Colored dot in left column
        auto leftCol = row.removeFromLeft(LEFT_COL_W);
        auto dotX = leftCol.getX() + 20.0f;
        auto dotY = leftCol.getCentreY() - 6.0f;
        g.setColour(juce::Colour(r.color));
        g.fillEllipse(dotX, dotY, 12.0f, 12.0f);

        // Stem name
        auto nameRect = leftCol.withLeft((int)dotX + 20);
        g.setFont(AR::font(AR::FontRole::label));
        g.setColour(juce::Colour(AR::FG));
        g.drawText(r.name, nameRect, juce::Justification::centredLeft);

        // Waveform in center column
        auto centerCol = row;  // already trimmed LEFT_COL_W; will also trim RIGHT_COL_W
        centerCol = centerCol.withTrimmedRight(RIGHT_COL_W);
        g.setColour(juce::Colour(AR::BG_DEEP));
        g.fillRect(centerCol.reduced(0, 8));

        if (r.thumb && r.thumb->getTotalLength() > 0.0) {
            float alpha = (*r.ctx_muted) ? 0.25f : 0.75f;
            g.setColour(juce::Colour(r.color).withAlpha(alpha));
            r.thumb->drawChannels(g, centerCol.reduced(0, 8),
                                  0.0, r.thumb->getTotalLength(), 0.8f);
        }

        // Unity tick on gain slider track (center = 1.0 on 0–2 range = 50%)
        auto rightCol = juce::Rectangle<int>(
            row.getRight() - RIGHT_COL_W, row.getY(), RIGHT_COL_W, row.getHeight());
        int sliderLeft  = rightCol.getX() + TOGGLE_SIZE * 2 + TOGGLE_GAP * 3;
        int sliderRight = rightCol.getRight() - 8;
        int tickX       = sliderLeft + (sliderRight - sliderLeft) / 2;
        g.setColour(juce::Colour(AR::COMMENT));
        g.fillRect(tickX - 1, row.getCentreY() - 6, 2, 12);
    }

    void layoutRow(juce::Rectangle<int> row, int i)
    {
        if (!rows_initialized_) return;
        auto& r = rows_[i];

        // Right column: [M] [S] [slider]
        auto rightCol = row.removeFromRight(RIGHT_COL_W).reduced(4, 12);

        // Gain slider first (right side of rightCol)
        auto sliderBounds = rightCol.removeFromRight(SLIDER_W);
        r.gain_slider.setBounds(sliderBounds);

        rightCol.removeFromRight(TOGGLE_GAP);

        // [S] then [M] (right to left)
        r.solo_btn.setBounds(rightCol.removeFromRight(TOGGLE_SIZE)
                                     .withSizeKeepingCentre(TOGGLE_SIZE, TOGGLE_SIZE));
        rightCol.removeFromRight(TOGGLE_GAP);
        r.mute_btn.setBounds(rightCol.removeFromRight(TOGGLE_SIZE)
                                     .withSizeKeepingCentre(TOGGLE_SIZE, TOGGLE_SIZE));
    }

    void handleMute(int idx)
    {
        auto& r = rows_[idx];
        *r.ctx_muted = r.mute_btn.getToggleState();
        repaint();
    }

    void handleSolo(int idx)
    {
        bool nowSolo = rows_[idx].solo_btn.getToggleState();
        *rows_[idx].ctx_solo = nowSolo;

        if (nowSolo) {
            // Mute all other stems
            for (int i = 0; i < 4; ++i) {
                if (i != idx) {
                    *rows_[i].ctx_muted = true;
                    rows_[i].mute_btn.setToggleState(true, juce::dontSendNotification);
                    *rows_[i].ctx_solo = false;
                    rows_[i].solo_btn.setToggleState(false, juce::dontSendNotification);
                }
            }
            // Unmute this stem
            *rows_[idx].ctx_muted = false;
            rows_[idx].mute_btn.setToggleState(false, juce::dontSendNotification);
        } else {
            // Clear all solos and mutes
            for (int i = 0; i < 4; ++i) {
                *rows_[i].ctx_solo  = false;
                *rows_[i].ctx_muted = false;
                rows_[i].solo_btn.setToggleState(false, juce::dontSendNotification);
                rows_[i].mute_btn.setToggleState(false, juce::dontSendNotification);
            }
        }
        repaint();
    }

    // ── Audio infrastructure ──────────────────────────────────────────────────
    juce::AudioFormatManager  format_manager_;
    juce::AudioThumbnailCache thumbnail_cache_;
    juce::AudioThumbnail      vocals_thumb_;
    juce::AudioThumbnail      drums_thumb_;
    juce::AudioThumbnail      bass_thumb_;
    juce::AudioThumbnail      other_thumb_;

    // ── Action bar ────────────────────────────────────────────────────────────
    juce::TextButton back_btn_;
    juce::TextButton next_btn_;

    JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(ScreenStemsReady)
};
```

**Implementation notes:**

1. `StemRowWidgets` contains four juce::TextButton and juce::Slider instances as value members (not pointers). They are added as child components in `initRows()` which is called from the constructor. JUCE Component semantics: `addAndMakeVisible` must be called from the constructor or onEnter, never in paint or resized.

2. The `initRows()` method is called from the constructor and uses pointer-to-member refs into `ctx_` so that reads/writes go directly to the shared ScreenContext fields.

3. Pointer members `ctx_gain`, `ctx_muted`, `ctx_solo` point into `ctx_` members. They are valid for the lifetime of ScreenStemsReady since ScreenContext is owned by the editor which outlives all screens.

4. The `rows_initialized_` flag guards against paintRowBackground/layoutRow being called before initRows completes (can happen if resized is called during construction on some platforms).

5. `std::array<StemRowWidgets, 4>` — `StemRowWidgets` has `juce::TextButton` and `juce::Slider` members which are not copyable. The array is default-constructed and `initRows()` configures each element in place. Because StemRowWidgets is non-copyable, avoid using `rows_ = {...}` initialization; use index-based setup in `initRows()` instead.

**Common issues:**

**Issue**: `std::array<StemRowWidgets, 4>` fails because StemRowWidgets is non-copyable (juce::TextButton is non-copyable).
**Fix**: Use `std::array` with default construction (zero-initializes POD fields). The array is declared in the class body and `initRows()` configures each element after default construction. Do NOT try to use aggregate initialization `{ {...}, {...} }`.

**Issue**: `ctx_.vocals_gain` etc. are `float` members of ScreenContext — taking `float*` to them is fine as long as ctx_ outlives the screen. Confirm by checking that ScreenContext is owned by the editor member (not a local) in PluginEditor.h.

**Issue**: `juce::Slider::LinearHorizontal` style in layoutRow — the slider needs to be added as a child component before setBounds is called. Confirm `addAndMakeVisible(r.gain_slider)` is called in `initRows()`, not in `layoutRow()`.

**Issue**: `juce::TextButton::buttonOnColourId` may need to be set with a custom LookAndFeel override for the button specifically. If the on-color isn't showing, use:
```cpp
r.mute_btn.setColour(juce::TextButton::buttonColourId,   juce::Colour(AR::ELEVATED));
r.mute_btn.setColour(juce::TextButton::buttonOnColourId, juce::Colour(AR::ERROR));
```

**Issue**: AudioThumbnail drawChannels draws nothing even though ctx_.stems.valid is true
**Fix**: The AudioFormatManager in ScreenStemsReady must call `registerBasicFormats()` — confirm it does. Also confirm the stem file paths are absolute and the files exist (they are written by ScreenSeparating's background thread).
